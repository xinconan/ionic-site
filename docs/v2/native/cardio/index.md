<h2>Usage</h2>

    (res: boolean) =&gt; {
      if(res){
        let options = {
          requireExpiry: true,
          requireCCV: false,
          requirePostalCode: false
        };
        CardIO.scan(options);
      }
    }
  );
</code></pre>
