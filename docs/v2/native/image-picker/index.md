<h2>Usage</h2>

<pre><code class="lang-typescript">import { ImagePicker } from &#39;ionic-native&#39;;



ImagePicker.getPictures(options).then((results) =&gt; {
  for (var i = 0; i &lt; results.length; i++) {
      console.log(&#39;Image URI: &#39; + results[i]);
  }
}, (err) =&gt; { });
</code></pre>
