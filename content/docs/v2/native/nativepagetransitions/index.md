<!-- description -->

<p>The Native Page Transitions plugin uses native hardware acceleration to animate your transitions between views. You have complete control over the type of transition, the duration, and direction.</p>


<!-- @platforms tag -->
<h2>Supported platforms</h2>

<ul>
  <li>iOS</li><li>Android</li><li>Windows Phone</li>
</ul>

<!-- @platforms tag end -->


<!-- @usage tag -->

<h2>Usage</h2>

<pre><code>import {NativePageTransitions, NativeTransitionOptions} from &#39;ionic-native&#39;;

let options: NativeTransitionOptions = {
   direction: &#39;up&#39;,
   duration: 500,
   slowdownfactor: 3,
   slidePixels: 20,
   iosdelay: 100,
   androiddelay: 150,
   winphonedelay: 250,
   fixedPixelsTop: 0,
   fixedPixelsBottom: 60
 };

NativePageTransitions.slide(options)
  .then(onSuccess)
  .catch(onError);
</code></pre>
